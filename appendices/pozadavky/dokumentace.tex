\section{Požadavky na dokumentaci}

\begin{OstatniPozadavkyTabulka}{DOK}{Požadavky na dokumentaci}
  \OstatniPozadavekRow{Dokumentace datových sad ve formě HTML}{Dokumentace každé publikované datové sady (struktura, význam polí, frekvence aktualizace, podmínky užití) musí být dostupná jako webová stránka v HTML, nikoliv jako stahovatelný PDF nebo Word dokument. Tisková podoba je řešena přes CSS, ne přes samostatný PDF dokument.}{1}
  \OstatniPozadavekRow{Strojově čitelný popis API jako součást dokumentace}{Dokumentace čtecího API musí obsahovat strojově čitelný popis (např. ve formátu OpenAPI nebo JSON Schema) i jeho lidsky čitelnou podobu generovanou ze stejného zdroje. Lidsky čitelná podoba je dostupná jako webová stránka (HTML), ne jako stahovatelný PDF dokument.}{1}
  \OstatniPozadavekRow{Kontextová nápověda v uživatelském rozhraní}{Uživatelské rozhraní musí poskytovat kontextovou nápovědu pro registrační formuláře (budova, místnost, senzor). Nápověda u každého formulářového pole vysvětluje, co se od uživatele očekává a proč je daný údaj sbírán. Souhrnná nápověda formou HTML stránek je dostupná z hlavní navigace aplikace.}{2}
\end{OstatniPozadavkyTabulka}
