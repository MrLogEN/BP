% Použití: \logentry{datum}{model}{soubory}{účel}{vstup}{výstup}{přínos}
\newcommand{\logentry}[7]{%
  \noindent\rule{\textwidth}{0.4pt}
  \begin{description}[leftmargin=!, labelwidth=3.5cm, font=\normalfont\bfseries]
    \item[Datum a čas:] #1
    \item[AI nástroj:] #2
    \item[Soubor(y):] \texttt{#3}
    \item[Účel použití:] #4
    \item[Vstupní data:] #5
    \item[Výstup AI:] #6
    \item[Vlastní přínos:] #7
  \end{description}
  \vspace{0.5em}
}
