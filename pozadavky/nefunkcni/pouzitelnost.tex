\subsection{Použitelnost}

\begin{OstatniPozadavkyTabulka}{POU}{Požadavky na použitelnost}
  \OstatniPozadavekRow{Vícejazyčné uživatelské rozhraní}{Uživatelské rozhraní aplikace musí být dostupné v češtině a angličtině. Volba jazyka je dostupná v každém okamžiku z rozhraní; preference uživatele je zapamatována mezi relacemi.}{1}
  \OstatniPozadavekRow{Přístupnost podle WCAG 2.1 úrovně AA}{Uživatelské rozhraní musí splňovat doporučení W3C Web Content Accessibility Guidelines verze 2.1 na úrovni AA, aby aplikace byla použitelná i osobami se zrakovým, sluchovým, motorickým nebo kognitivním omezením. Týká se zejména kontrastu textu, ovladatelnosti klávesnicí, popisků formulářových polí a strojově čitelné struktury stránek.}{2}
  \OstatniPozadavekRow{Responzivní uživatelské rozhraní}{Uživatelské rozhraní musí být použitelné na zařízeních se šířkou obrazovky od 360~px (mobilní telefon) do běžných stolních monitorů. Klíčové funkce (registrace entit, prohlížení dat, mapa) musí zůstat ovladatelné dotykem i myší.}{2}
\end{OstatniPozadavkyTabulka}
